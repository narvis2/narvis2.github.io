    // 2. DropDownMenu ë¥¼ Controller í•  ë²„íŠ¼ ë° Event ìƒì„±

        modifier = Modifier.wrapContentSize(),

        }) {
            Text(text = "Print Compose Dropdown")
        }
    }
}
